import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Given dictionary
animal_data = {
    'dog': {
        'lifespan': 12,
        'gestation_period': 65,
        'population_helsinki_approx': 781000
    },
    'cat': {
        'lifespan': 14,
        'gestation_period': 63,
        'population_helsinki_approx': 150000
    }
}

# Converting dictionary to DataFrame
animal_df = pd.DataFrame(animal_data).T

# Resetting index to have a column for animal types
animal_df.reset_index(inplace=True)

# Renaming columns for better understanding
animal_df.columns = ['Animal', 'Lifespan (years)', 'Gestation Period (days)', 'Population Helsinki Approx']

# Creating bar charts for each column
animal_df.plot(x='Animal', kind='bar', subplots=True, layout=(3,1), figsize=(10,20), legend=False)

# Adjusting the layout and displaying the plot
plt.tight_layout()
plt.show()
